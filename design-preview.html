  /* ── Accent — user-overridable ── */

  --fs-xs:  12px;   /* captions, tunnel label               */
  --fs-sm:  14px;   /* badges, secondary text               */
  --fs-md:  18px;   /* nav, body, labels                    */
  --fs-lg:  24px;   /* .btn control buttons                 */
  --fs-xl:  28px;   /* start-btn, filter-btn, sidebar h2    */
  --fs-2xl: 34px;   /* zoom buttons, content-panel h2       */
  --fs-3xl: 36px;   /* nav h1                               */

  /* ── Border radius (H2 fix: 5-stop scale) ── */
  --radius-sm:   8px;   /* buttons, badges, inputs          */
  --radius-md:   12px;  /* panels, cards (was 10/14/15px)   */
  --radius-lg:   20px;  /* tutorial overlay                 */
  --radius-pill: 36px;  /* filter buttons, slider columns   */
  --radius-full: 50%;   /* zoom buttons, thumbs             */

  /* ── Transitions ── */
